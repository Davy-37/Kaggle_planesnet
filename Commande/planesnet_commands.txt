# PlanesNet - commandes utiles
# (à lancer depuis la racine du projet, avec le json dans Data/planesnet/)

#  Réseaux de neurones (planesnet_cnn.py) 
# MLP
python planesnet_cnn.py --json Data/planesnet/planesnet.json --model mlp --epochs 20 --batch-size 256

# CNN
python planesnet_cnn.py --json Data/planesnet/planesnet.json --model cnn --epochs 25 --batch-size 256

# ResNet18 (avec amp si GPU)
python planesnet_cnn.py --json Data/planesnet/planesnet.json --model resnet18 --epochs 30 --batch-size 256 --amp

# Eval du meilleur checkpoint
python planesnet_cnn.py --json Data/planesnet/planesnet.json --evaluate --checkpoint runs_nn/best_model.pt


#  KNN 
python planesnet_knn.py --json Data/planesnet/planesnet.json

# avec recherche des meilleurs k / metric
python planesnet_knn.py --json Data/planesnet/planesnet.json --search --max-k 39


#  Naive Bayes 
python planesnet_bayes.py --json Data/planesnet/planesnet.json --pca 100


#  Arbre de décision 
python planesnet_decision_tree.py --json Data/planesnet/planesnet.json --export-graphviz
# pour afficher l'arbre en PNG (faut avoir graphviz installé) :
dot -Tpng runs_tree/tree.dot -o runs_tree/tree.png


#  Runner (tout en un) 
# train CNN
python planesnet_runner.py --algo cnn --mode train --json Data/planesnet/planesnet.json

# test
python planesnet_runner.py --algo cnn --mode test --json Data/planesnet/planesnet.json

# MLP ou ResNet
python planesnet_runner.py --algo mlp --mode train --json Data/planesnet/planesnet.json
python planesnet_runner.py --algo resnet18 --mode train --json Data/planesnet/planesnet.json --extra "--amp --epochs 30"

# lancer tout le supervisé d'un coup
python planesnet_runner.py --algo all-supervised --mode train --json Data/planesnet/planesnet.json
python planesnet_runner.py --algo all --mode train --json Data/planesnet/planesnet.json
